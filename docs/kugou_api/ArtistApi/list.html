Future<ArtistListResult> list({
  int type = 0,
  int sexType = 0,
  int musician = 0,
  int hotSize = 30,
}) async {
  return client.get<ArtistListResult>(
    '/ocean/v6/singer/list',
    params: {
      'musician': musician,
      'sextype': sexType,
      'showtype': 2,
      'type': type,
      'hotsize': hotSize,
    },
    encryptType: EncryptType.android,
    fromJson: (json) => ArtistListResult.fromJson(json),
  );
}
